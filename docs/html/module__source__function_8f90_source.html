202  subroutine ispli3(t,du,s)
203 !***********************************************************************************************************************************************************************************
204  implicit none
205 
206  real, intent(inout) :: s
207  real, intent(in) :: t
208  real, intent(in) :: du
209 
210  real :: dtsp
211  real :: abst
212  real :: t0
213 
214  t0 = du/2.0
215  dtsp = du/4.0
216 
217  abst = abs(t-t0)
218  if (abst >= 2.0*dtsp) then
219  s = 0.0
220  elseif ( (abst < (2.0*dtsp)) .and. (abst >= dtsp) ) then
221  s = ((2.0*dtsp-abst)**3)/(6.0*dtsp**3)
222  else
223  s = (3.0*abst**3 - 6.0*dtsp*abst**2 + 4.0*dtsp**3)/(6.0*dtsp**3)
224  endif
225  s = s/dtsp
226 
227  return
228 !***********************************************************************************************************************************************************************************
229  end subroutine ispli3

379  subroutine interp_linear( dim_num, data_num, t_data, p_data, interp_num, t_interp, p_interp )
380 !***********************************************************************************************************************************************************************************
381 
382  use mod_global_variables, only : error_stop
383 
384  implicit none
385 
386  integer, intent( in) :: data_num
387  integer, intent( in) :: dim_num
388  integer, intent( in) :: interp_num
389  real , intent( in) :: t_data(data_num)
390  real , intent( in) :: t_interp(interp_num)
391  real , intent( in) :: p_data(dim_num,data_num)
392  real , intent(out) :: p_interp(dim_num,interp_num)
393 
394  real :: t
395  integer :: interp
396  integer :: left
397  integer :: right
398 
399  character(len=255) :: info
400 
401  if ( .not. rvec_ascends_strictly( data_num, t_data ) ) then
402  write (info,'(a)') 'error in subroutine interp_linear: Independent variable array T_DATA is not strictly increasing.'
403  call error_stop(info)
404  endif
405 
406  do interp = 1, interp_num
407 
408  t = t_interp(interp)
409 !!
410 !! Find the interval [ TDATA(LEFT), TDATA(RIGHT) ] that contains, or is
411 !! nearest to, TVAL.
412 !!
413  call rvec_bracket( data_num, t_data, t, left, right )
414 
415  p_interp(1:dim_num,interp) = &
416  ( ( t_data(right) - t ) * p_data(1:dim_num,left) &
417  + ( t - t_data(left) ) * p_data(1:dim_num,right) ) &
418  / ( t_data(right) - t_data(left) )
419 
420  enddo
421 
422  return
423 !***********************************************************************************************************************************************************************************
424  end subroutine interp_linear
