135  real function lagrap(i,x,n)
136 !***********************************************************************************************************************************************************************************
137 
138  use mod_global_variables, only :&
139  rg_gll_abscissa&
140  ,rg_gll_abscissa_dist
141

143 
144  real , intent(in) :: x
145  integer, intent(in) :: i
146  integer, intent(in) :: n
147 
148  integer :: j
149 
150  lagrap = 1.0
151  do j = 1,n
152  if (j.ne.i) lagrap = lagrap*( (x-rg_gll_abscissa(j))*rg_gll_abscissa_dist(j,i) )

198  real function lagrad(i,x,n)
199 !***********************************************************************************************************************************************************************************
200 
201  use mod_global_variables, only : rg_gll_abscissa&
202  ,rg_gll_abscissa_dist
203 
204  implicit none
205 
206  real , intent(in) :: x
207  integer, intent(in) :: i
208  integer, intent(in) :: n
209 
210  real :: ltmp
211  real :: lprod
212  integer :: j
213  integer :: k
214 
215  lagrad = 0.0
216  lprod = 0.0
217  do j = 1,n
218  if (j.ne.i) then
219  lprod = 1.0
220  do k = 1,n
221  if (k.ne.i) then
222  if(k.eq.j) then
223  ltmp = rg_gll_abscissa_dist(j,i)
224  else
225  ltmp = (x-rg_gll_abscissa(k))*rg_gll_abscissa_dist(k,i)
226  endif
227  lprod = lprod*ltmp
228  endif
229  enddo
230  endif
231  lagrad = lagrad + lprod
232  lprod = 0.0
233  enddo
234 !
235 !***********************************************************************************************************************************************************************************
236  end function lagrad
